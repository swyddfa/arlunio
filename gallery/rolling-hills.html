import arlunio as ar
import numpy as np

@ar.shape
def Fill():
    return True

@ar.shape
def Cloud(x, y, *, x0=0, y0=0):
    r = 0.8
    x = x - x0
    y = y - y0 
    
    c1 = ar.S.Circle(r=r)
    c2 = ar.S.Circle(x0=1, y0=-0.2, r=(r*0.8))
    c3 = ar.S.Circle(x0=0.6, y0=0.2, r=(r*0.8))
    c4 = ar.S.Circle(y0=0.5, r=(r*0.8))
    
    return ar.any(
        c1(x=x, y=y),
        c2(x=np.abs(x), y=y),
        c3(x=np.abs(x), y=np.abs(y)),
        c4(x=x, y=y)
    )

image = Fill(color="aaeeff")
image += ar.S.Circle(x0=-1.2, y0=0.8, r=0.6, color="ffff00")

image += Cloud(x0=3.2, y0=2, scale=3, color="ffffff")
image += Cloud(x0=1, y0=2, scale=5, color="ffffff")
image += Cloud(x0=-2.7, y0=2, scale=3, color="ffffff")

image += ar.S.Circle(x0=-1, y0=-2.1, r=1.5, color="00cc00")
image += ar.S.Circle(x0=1, y0=-1.8, r=1.3, color="00ee00")
